SELECT
  Заказ.Ссылка         AS Заказ,
  Заказ.Номер          AS Номер,
  Заказ.Дата           AS Дата,
  Заказ.СуммаДокумента AS СуммаДокумента,
  Товар.НомерСтроки    AS НомерСтроки,
  Товар.Номенклатура   AS Номенклатура,
  Товар.Количество     AS Количество,
  Товар.Цена           AS Цена,
  Товар.Сумма          AS Сумма

FROM Документ.ЗаказКлиента AS Заказ

LEFT JOIN
  Документ.ЗаказКлиента.Товары AS Товар
ON Заказ.Ссылка = Товар.Ссылка

ORDER BY Заказ.Ссылка, Товар.НомерСтроки ASC
;
