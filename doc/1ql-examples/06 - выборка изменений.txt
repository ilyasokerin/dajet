SELECT TOP 10
  
  Изменения.УзелОбмена AS УзелОбмена,
  ПланОбмена.Код       AS КодУзлаПолучателя,
  Данные.Ссылка        AS Ссылка,
  Данные.Код           AS КодДанных,
  Данные.Наименование  AS Наименование

FROM Справочник.Номенклатура.Изменения AS Изменения

  INNER JOIN
    ПланОбмена.ПланОбменаДанными AS ПланОбмена
  ON Изменения.УзелОбмена = ПланОбмена.Ссылка

  LEFT JOIN
    Справочник.Номенклатура AS Данные
  ON Изменения.Ссылка = Данные.Ссылка

WHERE ПланОбмена.Код = 'N001'

ORDER BY Данные.Код
;
