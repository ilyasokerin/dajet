DECLARE @Документ     Документ.ЗаказКлиента   = "ba1f9927-0aec-11ed-9cd3-408d5c93cc8e";
DECLARE @ПустаяСсылка Справочник.Номенклатура = "00000000-0000-0000-0000-000000000000";
DECLARE @Номенклатура entity                  = {50:ba1f9927-0aec-11ed-9cd3-408d5c93cc8e};

SELECT
  Регистратор,
  Номенклатура
FROM
  РегистрНакопления.РегистрНакопленияОстатки

WHERE Регистратор = @Документ
  AND Номенклатура = @Номенклатура
  AND Номенклатура <> @ПустаяСсылка
;
