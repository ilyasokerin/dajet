CREATE NAMESPACE Документ;

CREATE TYPE Документ AS ENTITY
(
  Дата     datetime,
  Номер    string[10],
  Проведён boolean
);

CREATE TYPE Документ.ПоступлениеТовара AS Документ
(
  Поставщик      Справочник.Контрагенты,
  Получатель     Справочник.Организации,
  СкладТоваров   Справочник.МестаХранения,
  СуммаДокумента numeric(12,2)

  CREATE TYPE Товары AS ТабличнаяЧасть
  (
    Номенклатура Справочник.Номенклатура,
    Количество   numeric(4,0),
    Цена         numeric(6,2),
    Сумма        numeric(10,2)
  )
);

CREATE TABLE Документ.ПоступлениеТовара        OF Документ.ПоступлениеТовара;
CREATE TABLE Документ.ПоступлениеТовара.Товары OF Документ.ПоступлениеТовара.Товары;
