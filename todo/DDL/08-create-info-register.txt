CREATE NAMESPACE РегистрСведений;

CREATE TYPE РегистрСведений
(
  Период datetime PRIMARY KEY
);

CREATE TYPE РегистрСведений.КурсыВалют AS РегистрСведений
(
  Валюта    Справочник.Валюты PRIMARY KEY,
  Курс      numeric(12,4),
  Кратность numeric(4,0)
);

CREATE TABLE РегистрСведений.КурсыВалют OF РегистрСведений.КурсыВалют;
