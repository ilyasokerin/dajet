CREATE TYPE РегистрСведений
PRIMARY KEY Период,Регистратор,НомерЗаписи
(
  Период      datetime,
  Регистратор Документ,
  НомерЗаписи numeric(9,0),
  Активность  boolean
)

CREATE TYPE КурсыВалют AS РегистрСведений
DROP COLUMN Регистратор, НомерЗаписи, Активность
PRIMARY KEY Валюта
(
  Валюта    Справочник.Валюты,
  Курс      numeric(12,4),
  Кратность numeric(4,0)
)

CREATE TABLE РегистрСведений_КурсыВалют OF РегистрСведений.КурсыВалют
